"""
    創建老婆類，老婆列表。
    定義下列函數：
        1)獲取所有老婆姓名
        2)獲取顏值大於80的所有老婆物件
        3)獲取年齡最大的老婆對象
        4)根據顏值對老婆清單進行昇冪排列
"""

class Wife:
    def __init__(self, name="", face_score=0, age=0):
        self.name = name
        self.face_score = face_score
        self.age = age

    def print_self(self):
        print(self.name, self.face_score, self.age)

list_wife = [
    Wife("阿珂", 100, 23),
    Wife("蘇荃", 92, 32),
    Wife("雙兒", 90, 25),
    Wife("小郡主", 76, 22),
    Wife("方怡", 75, 27),
    Wife("建寧", 86, 25),
    Wife("曾柔", 67, 24),
]

# 1)獲取所有老婆姓名
def find01():
    result = []
    for item in list_wife:
        result.append(item.name)
    return result

print(find01())

# 2)獲取顏值大於80的所有老婆物件
def find02():
    result = []
    for item in list_wife:
        if item.face_score > 80:
            result.append(item)
    return result

for item in find02():
    item.print_self()

# 3)獲取年齡最大的老婆對象
def get_max():
    max_value = list_wife[0]
    for i in range(1, len(list_wife)):
        if max_value.age < list_wife[i].age:
            max_value = list_wife[i]
    return max_value

re = get_max()
re.print_self()

# 4)根據顏值對老婆清單進行昇冪排列
def order_by():
    for r in range(len(list_wife) - 1):
        for c in range(r + 1, len(list_wife)):
            if list_wife[r].face_score > list_wife[c].face_score:
                list_wife[r], list_wife[c] = list_wife[c], list_wife[r]

order_by()
print("-----------------")
for item in list_wife:
    item.print_self()
"""
    練習：對象計數器
        使用類變數記錄
        使用類方法列印
"""

class Wife:
    count = 0

    @classmethod
    def print_count(cls):
        print("總共娶了",cls.count,"個")

    def __init__(self):
        Wife.count += 1

w01 = Wife()
w02 = Wife()
w03 = Wife()
w04 = Wife()
w05 = Wife()
Wife.print_count()
"""
    創建敵人類,資料：姓名,血量(0--100),攻擊力(1--50)
"""

class Enemy:
    def __init__(self, name="", hp=0, atk=0):
        self.name = name
        self.hp = hp
        self.atk = atk

    @property
    def hp(self):
        return self.__hp

    @hp.setter
    def hp(self, value):
        if 1 <= value <= 100:
            self.__hp = value
        else:
            raise Exception("血量超過範圍")

    @property
    def atk(self):
        return self.__atk

    @atk.setter
    def atk(self, value):
        if 1 <= value <=50:
            self.__atk = value
        else:
            raise Exception("血量超過範圍")

e01 = Enemy("滅霸", 99, 50)
print(e01.hp)
print(e01.atk)
"""
    請以物件導向的思想，描述以下場景：
        玩家(攻擊力)攻擊敵人,敵人(血量)受傷後還可能死亡。
"""

class Player:
    def __init__(self, atk=0):
        self.atk = atk

    def attack(self, target):
        print("打你")
        target.damage(self.atk)

class Enemy:
    def __init__(self, hp=0):
        self.hp = hp

    def damage(self, value):
        print("呃！")
        self.hp -= value
        if self.hp <= 0:
            self.death()

    def death(self):
        print("死亡嘍")

p01 = Player(50)
e01 = Enemy(100)
p01.attack(e01)
p01.attack(e01)
"""
    請以物件導向的思想，描述以下場景：
        張無忌教趙敏九陽神功
        趙敏教張無忌化妝
        張無忌上班掙了10000元
        趙敏上班掙了8000元
"""
class Person:
    def __init__(self, name=""):
        self.name = name

    def teach(self,other,skill):
        print(self.name,"教",other.name,skill)

    def work(self,money):
        print(self.name,"工作掙了",money,"元")

zwj = Person("張無忌")
zm = Person("趙敏")
zwj.teach(zm,"九陽神功")
zm.teach(zwj,"化妝")
zwj.work(10000)
zm.work(8000)
"""
    定義父類
        動物(行為：吃)
    定義子類
        狗(行為：跑)
        鳥(行為：飛)
    體會isinstance/issubclass/type
"""
class Animal:
    def eat(self):
        print("吃飯飯嘍")

class Dog(Animal):
    def run(self):
        print("跑嘍")

class Bird(Animal):
    def fly(self):
        print("飛嘍")

d01 = Dog()
print(isinstance(d01,Dog))
print(isinstance(d01,Animal))

print(issubclass(Bird,Dog))

print(type(d01) == Animal)
"""
    定義父類
        車(數據：品牌,價格)
    定義子類
        電動車(數據:充電速度,電池容量)
    畫出記憶體圖
"""

class Car:
    def __init__(self, brand="", price=0):
        self.brand = brand
        self.price = price

class Electrocar(Car):
    def __init__(self, brand="", price=0, charging_speed=0, battery_capacity=0):
        super().__init__(brand,price)
        self.charging_speed = charging_speed
        self.battery_capacity = battery_capacity

e01 = Electrocar("寶馬",300000,60,10000000)
"""
    手雷爆炸,傷害玩家(扣血,碎屏)和敵人(扣血,頭頂爆字)。
    變化點：鴨子,房子...
    寫出體現三大特徵的代碼：
        封裝：根據需求分解出手雷類,玩家類,敵人類
        繼承：使用攻擊目標隔離手雷與玩家,敵人
        多態：手雷調用攻擊目標,在玩家和敵人物件中體現了不同效果
              玩家和敵人分別重寫了攻擊目標的受傷方法
"""

class Grenade:
    def explode(self, target):
        target.damage()

class AttackTarget:
    def damage(self):
        print("扣血")

# ---------------------
class Player(AttackTarget):
    def damage(self):
        super().damage()
        print("碎屏")

class Enemy(AttackTarget):
    def damage(self):
        super().damage()
        print("頭頂爆字")

g01 = Grenade()
p01 = Player()
e01 = Enemy()
g01.explode(e01)
"""
    創建圖形管理器：
        1. 記錄所有圖形
        2. 計算總面積
    圖形：
        1. 矩形
        2. 圓形
        ...
    要求：
        增加新圖形,管理器代碼不變.

    畫出設計圖
    寫出哪裡體現了四大原則
        開閉原則：增加新圖形,管理器代碼不變.
        單一職責：Rectanlge負責矩形面積的演算法，Circle負責圓形面積的演算法，
                GraphicManager負責統一管理圖形
        依賴倒置：圖形管理器沒有調用圓形、矩形演算法，而是調用圖形類。
        組合複用：圖形管理器與具體的各種圖形是組合關係。
"""

class GraphicManager:
    def __init__(self):
        self.__graphics = []

    def add_graphic(self, graphic):
        self.__graphics.append(graphic)

    def calculate_total_area(self):
        total_area = 0
        for item in self.__graphics:
            total_area += item.get_area()
        return total_area

class Graphic:
    def get_area(self):
        pass

# ------------------------
class Rectanlge(Graphic):
    def __init__(self, lenght=0, width=0):
        self.lenght = lenght
        self.width = width

    def get_area(self):
        return self.lenght * self.width

class Circle(Graphic):
    def __init__(self, radius=0):
        self.radius = radius

    def get_area(self):
        return 3.14 * self.radius ** 2

manager = GraphicManager()
manager.add_graphic(Rectanlge(2, 6))
manager.add_graphic(Circle(5))
print(manager.calculate_total_area())
